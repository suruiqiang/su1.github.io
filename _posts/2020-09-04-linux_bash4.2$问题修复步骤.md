# 解决方案
```
## 拷贝模版文件
># cd /etc/skel/  
># cp .bash_logout   /home/gpadmin
># cp .bash_profile  /home/gpadmin
># cp .bashrc        /home/gpadmin
># cd /home/gpadmin
># chown -R gpadmin.gpadmin .bash_logout
># chown -R gpadmin.gpadmin .bash_profile
># chown -R gpadmin.gpadmin .bashrc

## 配置 .bashrc
```
